import random
from questionaire import *

# List of question functions
question = [first, second, third, fourth, fifth, sixth, seventh, eighth, ninth, tenth]

# Shuffle the question list randomly
random.shuffle(question)

quest_count = 0
valid_answer = 0
invalid_answer = 0
score = 0

# Keep track of used indexes for clarity (optional)
asked_questions = []

while quest_count < 10:
    try:
        user_choice = int(input('Pick a question number from (1-10): '))

        if user_choice < 1 or user_choice > 10:
            print('Invalid number. Please choose between 1 and 10.')
            continue

        index = user_choice - 1

        if index in asked_questions:
            print('You have already answered this question. Choose another one.')
            continue

        asked_questions.append(index)

        # Call the corresponding shuffled question function
        result = question_list[index](True)

        if result:
            print("‚úÖ Correct!\n")
            valid_answer += 1
            score += 1
        else:
            print("‚ùå Wrong answer or max attempts reached.\n")
            invalid_answer += 1

        quest_count += 1

    except ValueError:
        print("Please enter a valid number.")
        continue

# Final score summary
print("üéâ All done!")
print(f"You got {valid_answer} correct, and {invalid_answer} incorrect.")
print(f"Your total score is: {score}/10")
